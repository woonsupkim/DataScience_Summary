#data
hnobs = 100
x_train = np.vstack((\
        np.random.default_rng(1).normal(1,1, size = (hnobs,2)),\
        np.random.default_rng(3).normal(-1,1, size = (hnobs,2))\
        ))
y_train = np.append(np.full(hnobs,'A'), np.full(hnobs,'B'))

x_test = np.array([[0, 3], [-4, -5], [2, 0]])
y_test = np.array(['A', 'B', 'A'])

plt.scatter(x_train[:,0], x_train[:,1], c = (y_train == 'A'),\
            s = 25, alpha = 0.5)
plt.scatter(x_test[:,0], x_test[:,1], marker = '*', \
            c = (y_test == 'A')  , s = 300, alpha = 1)
plt.xlabel('X1')
plt.ylabel('X2')
plt.show()
