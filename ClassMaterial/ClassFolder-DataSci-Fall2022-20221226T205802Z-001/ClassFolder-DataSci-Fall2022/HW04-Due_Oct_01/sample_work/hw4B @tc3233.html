#for this two tree, we calculate the sum error, and then the R-square to compare them
ESS_sex = np.sum((df3['height'] - tree_sex.predict(np.array(df3['male']).reshape(-1,1)))**2)
ESS_weight = np.sum((df3['height'] - tree_weight.predict(np.array(df3['weight']).reshape(-1,1)))**2)
TSS = np.sum((df3['height'] - np.mean(df3['height']))**2)
#Calculate the Rsquare in order to compare two models
Rsquare_sex = 1 - ESS_sex/TSS
Rsquare_weight = 1 - ESS_weight/TSS
