def mean(x):
    sums = 0
    for i in x:
        sums += i
    return(sums/len(x))

def mse(y1, y2):
    n = len(y1)
    sum_square = 0
    for i in range(len(y1)):
        sum_square += (y1[i] - y2[i])**2
    return sum_square/n

k_range = range(1,11)
MSE = []
for i in k_range:
    MSE_k = []
    #attention! The iteration would only loop once, we have to reset this in every loop.
    cvsplit = KFold(n_splits = 6, random_state = 1, shuffle = True).split(datax)
    for m,n in cvsplit:
        x_train = [datax[t] for t in m]
        x_test = [datax[t] for t in n]
        y_train = [datay[t] for t in m]
        y_test = [datay[t] for t in n]
        knn_res = KNeighborsRegressor(i).fit(x_train, y_train)
        y_pred = knn_res.predict(x_test)
        MSE_kn = mse(y_test, y_pred)
        MSE_k.append(MSE_kn)
    MSE.append(mean(MSE_k))
